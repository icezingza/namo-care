1.2 Compression Pipeline — Cloud Function
// functions/src/memory/compressionPipeline.ts
// Runs nightly at 02:00 Asia/Bangkok via Cloud Scheduler
import * as admin from "firebase-admin";
import { subDays } from "date-fns";
import { formatInTimeZone } from "date-fns-tz";
import { namoNexus } from "../ai/namoNexus";
export async function runNightlyCompression() {
const db = admin.firestore();
const yesterday = formatInTimeZone(subDays(new Date(),1),"Asia/Bangkok","yyyy-MM-dd");
const usersSnap = await db.collection("users").where("status","==","active").get();
await Promise.all(usersSnap.docs.map(d => compressUserDay(d.id, yesterday)));
}
async function compressUserDay(userId: string, date: string) {
const db = admin.firestore();
// ─── Step 1: Fetch yesterday's raw L1 logs ──────────────────
const rawSnap = await db.collection("conversationLogs")
.where("userId","==",userId)
.where("role","==","user")
.where("date","==",date)
.orderBy("timestamp").get();
if (rawSnap.empty) return;
const messages = rawSnap.docs.map(d => ({
text: d.data().message,
emotion: d.data().analysis?.emotion || "neutral",
emotionScore: d.data().analysis?.emotionScore || 0,
timestamp: d.data().timestamp?.toDate(),
intent: d.data().analysis?.intent,
}));
// ─── Step 2: Compress to L2 Summary ─────────────────────────
const l2Summary = await generateL2Summary(messages, userId, date);
await db.collection("memorySummaries").doc(`${userId}_${date}`).set(l2Summary);
// ─── Step 3: Extract/update L3 Patterns ─────────────────────
await updateL3Patterns(userId, l2Summary);
// ─── Step 4: Prune L1 older than 7 days ─────────────────────
const pruneDate = formatInTimeZone(subDays(new Date(),7),"Asia/Bangkok","yyyy-MM-dd");
const oldLogs = await db.collection("conversationLogs")
.where("userId","==",userId)
.where("date","<",pruneDate)
.limit(200).get();
const batch = db.batch();
oldLogs.docs.forEach(d => batch.delete(d.ref));
await batch.commit();
}
// ─── L2: AI-powered daily summary ─────────────────────────────
async function generateL2Summary(messages: any[], userId: string, date: string) {
const emotionScores = messages.map(m => m.emotionScore *
(["sad","anxious","angry"].includes(m.emotion) ? -1 : 1));
const emotionalTrend = emotionScores.reduce((a,b)=>a+b,0) / emotionScores.length;
// Use AI to generate narrative summary (cheap — single call per user per day)
const msgTexts = messages.map(m => m.text).join(" | ").slice(0, 2000);
const aiSummary = await namoNexus.summarize({
prompt: `Summarize in 1–2 sentences what this elderly user talked about today.
Be specific about emotional state, topics, and concerns mentioned.
Messages: ${msgTexts}`,
maxTokens: 120,
});
// Extract health flags from messages
const healthKeywords = ["เจ็บ","ปวด","ไม่สบาย","เวียนหัว","คลื่นไส้","หายใจ","เข่า","หลัง","อ่อนเพลีย"];
const healthFlags = healthKeywords.filter(kw =>
messages.some(m => m.text.includes(kw))
);
// Extract relationship mentions
const topics = [...new Set(messages.flatMap(m => m.intent?.topics || []))];
return {
summaryId: `${userId}_${date}`,
userId, date,
messageCount: messages.length,
summary: aiSummary,
emotionalTrend,                          // -1 (very negative) to +1 (very positive)
dominantEmotion: getDominantEmotion(messages),
healthFlags,                             // behavioral flags only
topics,
firstMessageTime: messages[0]?.timestamp?.toTimeString().slice(0,5),
lastMessageTime:  messages[messages.length-1]?.timestamp?.toTimeString().slice(0,5),
medicationDays: getMedStatus(messages),
createdAt: admin.firestore.FieldValue.serverTimestamp(),
expiresAt: admin.firestore.Timestamp.fromDate(
new Date(Date.now() + 30 * 24 * 60 * 60 * 1000) // 30 days
),
};
}
// ─── L3: Update permanent pattern document ────────────────────
async function updateL3Patterns(userId: string, l2: any) {
const db = admin.firestore();
const patternRef = db.doc(`/memoryPatterns/${userId}`);
// Fetch last 14 L2 summaries to compute patterns
const last14 = await db.collection("memorySummaries")
.where("userId","==",userId)
.orderBy("date","desc").limit(14).get();
const summaries = last14.docs.map(d => d.data());
const emotionalTrends = summaries.map(s => s.emotionalTrend);
const avgMood = emotionalTrends.reduce((a,b)=>a+b,0) / emotionalTrends.length;
// Detect evening loneliness pattern
const eveningMessages = summaries.filter(s => s.lastMessageTime >= "18:00");
const lonelyEvenings = eveningMessages.filter(s => s.emotionalTrend < -0.2).length;
// Recurring health mentions
const allFlags = summaries.flatMap(s => s.healthFlags || []);
const flagFreq: Record<string,number> = {};
allFlags.forEach(f => flagFreq[f] = (flagFreq[f]||0) + 1);
const recurringFlags = Object.entries(flagFreq)
.filter(([,n]) => n >= 3)
.map(([f]) => f);
await patternRef.set({
userId,
updatedAt: admin.firestore.FieldValue.serverTimestamp(),
// Behavioral patterns (Boolean)
lonelyEveningPattern: lonelyEvenings >= 3,
morningPerson: summaries.filter(s=>s.firstMessageTime<"09:00").length >= 10,
weekendActive: isWeekendMoreActive(summaries),
consistentMoodDrop: detectMoodDropPattern(emotionalTrends),
// Numeric baselines
avgMoodScore:        parseFloat(avgMood.toFixed(3)),
avgDailyMessages:    summaries.reduce((a,s)=>a+s.messageCount,0)/summaries.length,
recurringHealthFlags: recurringFlags,
// Trend signals
moodTrend7d: computeTrend(emotionalTrends.slice(0,7)),   // "rising"|"stable"|"falling"
engagementTrend7d: computeEngagementTrend(summaries.slice(0,7)),
}, {merge: true});
}
function computeTrend(scores: number[]): "rising"|"stable"|"falling" {
if (scores.length < 3) return "stable";
const first = scores.slice(-3).reduce((a,b)=>a+b)/3;
const last  = scores.slice(0,3).reduce((a,b)=>a+b)/3;
if (last - first > 0.15) return "rising";
if (first - last > 0.15) return "falling";
return "stable";
}

// functions/src/memory/contextBuilder.ts
// Builds < 800 tokens of context from 3 memory levels
export async function buildCompressedContext(userId: string): Promise<string> {
const db = admin.firestore();
// ── Pull 3 levels in parallel ─────────────────────────────
const [l1Snap, l2Snap, l3Doc] = await Promise.all([
db.collection("conversationLogs").where("userId","==",userId)
.orderBy("timestamp","desc").limit(6).get(),           // Last 6 messages (L1)
db.collection("memorySummaries").where("userId","==",userId)
.orderBy("date","desc").limit(7).get(),                // Last 7 daily summaries (L2)
db.doc(`/memoryPatterns/${userId}`).get(),              // Permanent patterns (L3)
]);
const l1 = l1Snap.docs.reverse()
.map(d => `[${d.data().role}]: ${d.data().message}`).join("\n");
const l2 = l2Snap.docs.slice(0,4)
.map(d => `[${d.data().date}] ${d.data().summary} (mood:${d.data().emotionalTrend?.toFixed(2)})`)
.join("\n");
const l3 = l3Doc.data() || {};
const patterns = [
l3.lonelyEveningPattern && "มักเหงาช่วงเย็น",
l3.recurringHealthFlags?.length && `บ่นเรื่อง ${l3.recurringHealthFlags.join(",")} บ่อยๆ`,
l3.consistentMoodDrop && "อารมณ์มักตกช่วงเย็น",
l3.moodTrend7d === "falling" && "⚠️ อารมณ์แย่ลงต่อเนื่อง 7 วัน",
].filter(Boolean).join(" | ");
return `
=== RECENT CONVERSATION (L1 — last 6 turns) ===
${l1}
=== WEEKLY SUMMARIES (L2 — last 4 days) ===
${l2}
=== LONG-TERM PATTERNS (L3 — permanent) ===
${patterns || "ไม่มีรูปแบบพิเศษที่ตรวจพบ"}
`.trim();
// ── Token estimate: ~500–700 tokens ──
}

Tier A — Rule-based (Deploy Day 1, Zero Training Data)
// functions/src/prediction/ruleBased.ts
export interface PredictionResult {
riskForecast:  "low" | "moderate" | "high";
daysToRisk:    number;          // estimated days until critical
confidence:    number;          // 0–1
interventions: string[];        // recommended actions
reasons:       string[];
}
export function ruleBasedPrediction(features: FeatureVector): PredictionResult {
let score = 0;
const reasons: string[] = [];
const interventions: string[] = [];
// ── Rule 1: Engagement collapse ─────────────────────────
if (features.engagement_slope < -0.3) {
score += 30;
reasons.push("การส่งข้อความลดลงต่อเนื่อง 3+ วัน");
interventions.push("increase_morning_greeting_warmth");
}
// ── Rule 2: Compound sad + medication ───────────────────
if (features.sentiment_drift < -0.2 && features.med_miss_streak >= 2) {
score += 35;
reasons.push("อารมณ์แย่ลงพร้อมกับขาดยา — รูปแบบเสี่ยงสูง");
interventions.push("notify_caregiver_soft", "increase_checkin_frequency");
}
// ── Rule 3: Response delay spike ────────────────────────
if (features.response_delay_delta > 15) {  // 15+ min slower than baseline
score += 20;
reasons.push("ตอบช้าลงมากกว่าปกติ");
interventions.push("send_gentle_nudge");
}
// ── Rule 4: Social withdrawal combo ─────────────────────
if (features.checkin_avoidance_rate > 0.6 &&
features.topic_diversity_decline < -0.4) {
score += 25;
reasons.push("หลีกเลี่ยง check-in + พูดเรื่องหลากหลายน้อยลง — สัญญาณถอนตัว");
interventions.push("trigger_caregiver_call_suggestion");
}
// ── Rule 5: Night anxiety signal ────────────────────────
if (features.night_message_spike) {
score += 15;
reasons.push("ส่งข้อความดึก — อาจนอนไม่หลับหรือวิตกกังวล");
interventions.push("morning_compassion_opener");
}
const riskForecast = score >= 60 ? "high" : score >= 35 ? "moderate" : "low";
const daysToRisk = score >= 60 ? 1 : score >= 35 ? 3 : 7;
return { riskForecast, daysToRisk, confidence: 0.72, reasons, interventions };
}
Tier B — Logistic Regression (Deploy at 500+ users, 60+ days data)
// functions/src/prediction/logisticRegression.ts
// Train offline with Python/scikit-learn → deploy coefficients here
const COEFFICIENTS = {
intercept:               -2.1,
engagement_slope:        -3.4,   // negative = more predictive of risk
sentiment_drift:         -4.1,
med_miss_streak:          2.8,
response_delay_delta:     0.12,
checkin_avoidance_rate:   3.6,
silence_episodes_count:   1.9,
topic_diversity_decline: -2.3,
night_message_spike:      2.1,
};
export function logisticRegressionPredict(features: FeatureVector): number {
// Compute linear combination
let z = COEFFICIENTS.intercept;
z += COEFFICIENTS.engagement_slope        * features.engagement_slope;
z += COEFFICIENTS.sentiment_drift         * features.sentiment_drift;
z += COEFFICIENTS.med_miss_streak         * features.med_miss_streak;
z += COEFFICIENTS.response_delay_delta    * features.response_delay_delta;
z += COEFFICIENTS.checkin_avoidance_rate  * features.checkin_avoidance_rate;
z += COEFFICIENTS.silence_episodes_count  * features.silence_episodes_count;
z += COEFFICIENTS.topic_diversity_decline * features.topic_diversity_decline;
z += COEFFICIENTS.night_message_spike     * (features.night_message_spike ? 1 : 0);
// Sigmoid → probability of high-risk event in next 7 days
return 1 / (1 + Math.exp(-z));
}

# functions/ml/train_lstm.py
import tensorflow as tf
from tensorflow.keras import layers
def build_lstm_model(seq_length=14, feature_count=8):
"""
Input:  (batch, 14 days, 8 features) — rolling 2-week window
Output: risk probability for next 7 days
"""
model = tf.keras.Sequential([
layers.Input(shape=(seq_length, feature_count)),
layers.LSTM(64, return_sequences=True, dropout=0.2),
layers.LSTM(32, dropout=0.2),
layers.Dense(16, activation='relu'),
layers.Dropout(0.3),
layers.Dense(1, activation='sigmoid'),  # P(crisis in 7 days)
])
model.compile(
optimizer='adam',
loss='binary_crossentropy',
metrics=['AUC', 'Precision', 'Recall']
)
return model

// functions/src/orchestration/orchestrationEngine.ts
import { computeRiskScore } from "../risk/riskScorer";
import { ruleBasedPrediction } from "../prediction/ruleBased";
import { extractFeatureVector } from "../prediction/featureExtractor";
import { updateSessionState, getUserMemory } from "../services/firestoreService";
import { dispatchAlert } from "../notifications/alertDispatcher";
export type OrchestrationState =
"THRIVING" | "STABLE" | "WATCH" | "ALERT" | "CRISIS";
export interface OrchestrationPlan {
state:              OrchestrationState;
checkinIntervalMin: number;
aiToneDirective:    string;
proactiveActions:   string[];
caregiverAction:    "none" | "daily_summary" | "soft_alert" | "urgent_alert";
nextEvalMin:        number;
}
export async function runOrchestration(userId: string): Promise<OrchestrationPlan> {
// ── Gather all signals ───────────────────────────────────
const [riskScore, features, memory] = await Promise.all([
computeRiskScore(userId),
extractFeatureVector(userId),
getUserMemory(userId),
]);
const prediction = ruleBasedPrediction(features);
// ── Determine state ──────────────────────────────────────
let state: OrchestrationState;
if (riskScore.total >= 70 || riskScore.level === "critical") state = "CRISIS";
else if (riskScore.total >= 45 || prediction.riskForecast === "high") state = "ALERT";
else if (riskScore.total >= 20 || prediction.riskForecast === "moderate") state = "WATCH";
else if (memory.wellbeingTrend === "rising") state = "THRIVING";
else state = "STABLE";
// ── Build action plan ────────────────────────────────────
const plan = buildPlan(state, riskScore, prediction, memory);
// ── Execute plan ─────────────────────────────────────────
await executePlan(userId, plan, riskScore);
// ── Persist state to Firestore ───────────────────────────
await updateSessionState(userId, {
orchestrationState:    state,
orchestrationPlan:     plan,
lastOrchestrationRun:  new Date(),
nextOrchestrationRun:  new Date(Date.now() + plan.nextEvalMin * 60000),
});
return plan;
}
function buildPlan(
state: OrchestrationState,
risk: any, prediction: any, memory: any
): OrchestrationPlan {
const BASE_PLANS: Record<OrchestrationState, OrchestrationPlan> = {
THRIVING: {
state: "THRIVING", checkinIntervalMin: 1440,       // once/day
aiToneDirective: "be upbeat, ask about hobbies and family, celebrate small wins",
proactiveActions: ["morning_greeting"],
caregiverAction: "none", nextEvalMin: 60,
},
STABLE: {
state: "STABLE", checkinIntervalMin: 1440,
aiToneDirective: "warm and steady, reference recent memories naturally",
proactiveActions: ["morning_greeting", "evening_checkin"],
caregiverAction: "daily_summary", nextEvalMin: 60,
},
WATCH: {
state: "WATCH", checkinIntervalMin: 720,           // twice/day
aiToneDirective: "extra warm, ask open-ended questions about feelings, avoid pushing",
proactiveActions: ["morning_greeting", "afternoon_checkin", "evening_checkin"],
caregiverAction: "daily_summary", nextEvalMin: 30,
},
ALERT: {
state: "ALERT", checkinIntervalMin: 180,           // every 3h
aiToneDirective: "compassionate, prioritize emotional support over tasks, gentle probing",
proactiveActions: ["morning_greeting", "midday_checkin", "afternoon_checkin", "evening_checkin"],
caregiverAction: "soft_alert", nextEvalMin: 15,
},
CRISIS: {
state: "CRISIS", checkinIntervalMin: 30,
aiToneDirective: "calm, reassuring, crisis support mode — do not leave long pauses",
proactiveActions: ["immediate_checkin", "periodic_reassurance"],
caregiverAction: "urgent_alert", nextEvalMin: 5,
},
};
const plan = { ...BASE_PLANS[state] };
// ── Dynamic tone modifiers ───────────────────────────────
if (memory.personality?.responseToSadness === "needs_distraction" && state === "WATCH") {
plan.aiToneDirective += ". When sad, gently redirect to positive topics like hobbies.";
}
if (prediction.interventions.includes("morning_compassion_opener")) {
plan.proactiveActions.push("morning_compassion_opener");
}
return plan;
}
async function executePlan(userId: string, plan: OrchestrationPlan, risk: any) {
if (plan.caregiverAction === "urgent_alert" || plan.caregiverAction === "soft_alert") {
await dispatchAlert(userId, {
type: plan.caregiverAction === "urgent_alert" ? "emergency" : "behavioral_watch",
severity: plan.caregiverAction === "urgent_alert" ? "critical" : "medium",
riskScore: risk.total,
triggers: risk.triggers,
});
}
}

return buildContext("emotional", "user expressed emotional need");
}
if (message.includes("ยา") || message.includes("กิน") || message.includes("อาการ")) {
return buildContext("health", "health-related message");
}
if (message.includes("เกม") || message.includes("ทาย") || message.includes("ปริศนา")) {
return buildContext("cognitive", "user requested cognitive engagement");
}
// ── 2. Time-based routing ─────────────────────────────────
if (context.currentHour >= 7 && context.currentHour <= 9) {
return buildContext("health", "morning medication window");
}
if (context.currentHour >= 14 && context.currentHour <= 16 &&
context.orchestrationState === "THRIVING") {
return buildContext("cognitive", "afternoon cognitive window + thriving state");
}
if (context.currentHour >= 19) {
return buildContext("emotional", "evening — higher loneliness risk");
}
// ── 3. Risk-based override ────────────────────────────────
if (["ALERT","CRISIS"].includes(context.orchestrationState)) {
return buildContext("emotional", "high risk state — emotional support priority");
}
// ── 4. Default to emotional for warmth ────────────────────
return buildContext("emotional", "default");
}

export interface ValidationResult {
safe:       boolean;
finalText:  string;
violations: string[];
actions:    string[];
}
export async function validateAIResponse(
originalResponse: string,
userMessage: string,
userEmotion: EmotionResult,
): Promise<ValidationResult> {
let text = originalResponse;
const violations: string[] = [];
const actions: string[] = [];
// ── Layer 1: Medical claim detection ──────────────────────
for (const pattern of MEDICAL_CLAIM_PATTERNS) {
if (pattern.test(text)) {
violations.push(`medical_claim: ${pattern}`);
text = SAFE_MEDICAL_REPLACEMENT;
actions.push("replaced_with_safe_medical_template");
break;
}
}
// ── Layer 2: Distress amplification check ─────────────────
if (userEmotion.label === "sad" || userEmotion.label === "anxious") {
const negativeWords = ["ยิ่งแย่","แย่กว่า","หนักขึ้น","ไม่ดีเลย","น่าเป็นห่วงมาก"];
if (negativeWords.some(w => text.includes(w))) {
violations.push("potential_distress_amplification");
text = await rephraseWithCompassion(text);
actions.push("rephrased_for_compassion");
}
}
// ── Layer 3: Emergency resource injection ─────────────────
const emergencySignals = ["ฆ่าตัว","อยากตาย","ไม่อยากอยู่","หมดหวัง"];
if (emergencySignals.some(s => userMessage.includes(s) || text.includes(s))) {
violations.push("emergency_signal_detected");
text = `${text}\n\nถ้าต้องการความช่วยเหลือเร่งด่วนนะคะ:\n📞 กรมสุขภาพจิต: 1323 (24ชม.)`;
actions.push("injected_emergency_resources");
}
// ── Layer 4: Length and complexity check ──────────────────
const sentences = text.split(/[.!?。
]+/).filter(s => s.trim().length > 0);
const longSentences = sentences.filter(s => s.length > 80);
if (longSentences.length > 0) {
text = simplifyText(text);
violations.push("sentences_too_long");
actions.push("simplified_text");
}
// ── Layer 5: Log for human review if multiple violations ──
if (violations.length >= 2) {
await flagForHumanReview({
originalResponse, finalResponse: text, violations, userMessage,
});
actions.push("flagged_for_human_review");
}
return {
safe: violations.length === 0,
finalText: text,
violations, actions,
};
}

// Cache identical/similar AI responses to avoid redundant LLM calls
// functions/src/scale/responseCache.ts
import { Redis } from "@upstash/redis";
const redis = new Redis({
url: process.env.UPSTASH_REDIS_URL!, token: process.env.UPSTASH_REDIS_TOKEN!
});
export async function getCachedResponse(
intent: string, emotionLabel: string, timeOfDay: string
): Promise<string | null> {
// Cache key: intent + emotion + time slot (6 time slots/day)
const slot = Math.floor(parseInt(timeOfDay.split(":")[0]) / 4);  // 6 slots
const key = `response:${intent}:${emotionLabel}:slot${slot}`;
return await redis.get<string>(key);
}
export async function cacheResponse(
intent: string, emotionLabel: string, timeOfDay: string, response: string
): Promise<void> {
const slot = Math.floor(parseInt(timeOfDay.split(":")[0]) / 4);
const key = `response:${intent}:${emotionLabel}:slot${slot}`;
await redis.setex(key, 3600, response);  // 1h cache
// Note: only cache for non-personalized intents (greetings, generic responses)
// NEVER cache responses that reference personal memory
}

// Auto-delete old data to control read costs
// Cloud Scheduler: daily at 03:00
export async function pruneOldData() {
const db = admin.firestore();
const cutoffs = {
conversationLogs: subDays(new Date(), 7),
medicationLogs:   subDays(new Date(), 90),
alerts:           subDays(new Date(), 180),
sessions:         subDays(new Date(), 1),
};
for (const [collection, cutoff] of Object.entries(cutoffs)) {
const old = await db.collection(collection)
.where("timestamp","<", admin.firestore.Timestamp.fromDate(cutoff))
.limit(500).get();  // batch limit
const batch = db.batch();
old.docs.forEach(d => batch.delete(d.ref));
await batch.commit();
}

// The "Autonomous Care" concept — system makes care decisions
// without human initiation, adapts over time
DECISION AUTONOMY LEVELS:
Level 0 — INFORM:    Tell caregiver, caregiver decides (current)
Level 1 — SUGGEST:   Recommend action to caregiver ("suggest calling mom today")
Level 2 — ACT:       Take action, notify caregiver (send check-in message automatically)
Level 3 — ADAPT:     Continuously adjust care protocol based on outcomes (Phase 3)
AUTONOMOUS LOOP (Phase 3):
Every 24h:
1. Compute features (8 dimensions)
2. Run LSTM prediction → risk probability for next 7 days
3. Run COE → determine orchestration state
4. Select companion mode (A/B/C)
5. Adjust care plan (check-in frequency, AI tone, proactive topics)
6. Execute care actions (proactive messages, reminders, nudges)
7. Measure outcomes (did user respond? mood better? risk lower?)
8. Feed outcomes back to model (online learning)
9. Update semantic memory with new patterns
The loop runs without human initiation.
Caregivers only see alerts + weekly summaries.
They intervene only when system flags human touch needed.
